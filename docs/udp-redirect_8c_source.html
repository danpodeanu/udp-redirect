648static int parse_addr(const char *str, int port, struct sockaddr_storage *out) {
649 struct sockaddr_in *a4 = (struct sockaddr_in *)out;
650 struct sockaddr_in6 *a6 = (struct sockaddr_in6 *)out;
651
652 memset(out, 0, sizeof(*out));
653
654 if (inet_pton(AF_INET, str, &a4->sin_addr) == 1) {
655 out->ss_family = AF_INET;
656 a4->sin_port = htons(port);
657 return AF_INET;
658 }
659 if (inet_pton(AF_INET6, str, &a6->sin6_addr) == 1) {
660 out->ss_family = AF_INET6;
661 a6->sin6_port = htons(port);
662 return AF_INET6;
663 }
664 return -1;
665}

671static const char *addr_tostring(const struct sockaddr_storage *sa, char *buf, size_t len) {
672 const void *ptr;
673 if (sa->ss_family == AF_INET6)
674 ptr = &((const struct sockaddr_in6 *)sa)->sin6_addr;
675 else
676 ptr = &((const struct sockaddr_in *)sa)->sin_addr;
677 return inet_ntop((int)sa->ss_family, ptr, buf, (socklen_t)len) ? buf : "?";
678}

683static int addr_port(const struct sockaddr_storage *sa) {
684 if (sa->ss_family == AF_INET6)
685 return ntohs(((const struct sockaddr_in6 *)sa)->sin6_port);
686 return ntohs(((const struct sockaddr_in *)sa)->sin_port);
687}

693 return (sa->ss_family == AF_INET6)
694 ? (socklen_t)sizeof(struct sockaddr_in6)
695 : (socklen_t)sizeof(struct sockaddr_in);

708static int addr_equal(const struct sockaddr_storage *a, const struct sockaddr_storage *b) {
709 if (a->ss_family != b->ss_family)
710 return 0;
711 if (a->ss_family == AF_INET) {
712 const struct sockaddr_in *a4 = (const struct sockaddr_in *)a;
713 const struct sockaddr_in *b4 = (const struct sockaddr_in *)b;
714 return a4->sin_addr.s_addr == b4->sin_addr.s_addr &&
715 a4->sin_port == b4->sin_port;
716 }
717 if (a->ss_family == AF_INET6) {
718 const struct sockaddr_in6 *a6 = (const struct sockaddr_in6 *)a;
719 const struct sockaddr_in6 *b6 = (const struct sockaddr_in6 *)b;
720 return memcmp(&a6->sin6_addr, &b6->sin6_addr, sizeof(struct in6_addr)) == 0 &&
721 a6->sin6_port == b6->sin6_port;
722 }
723 return 0;
724}

751 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "%s address invalid: %s", desc, xaddr);
752
753 exit(EXIT_FAILURE);
754 }
755 family = (int)addr.ss_family;
756 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to address %s", desc, xaddr);
757 } else {
758 family = xfamily;
759 addr.ss_family = (sa_family_t)family;
760 if (family == AF_INET6) {
761 ((struct sockaddr_in6 *)&addr)->sin6_addr = in6addr_any;
762 ((struct sockaddr_in6 *)&addr)->sin6_port = htons(xport);
763 } else {
764 ((struct sockaddr_in *)&addr)->sin_addr.s_addr = INADDR_ANY;
765 ((struct sockaddr_in *)&addr)->sin_port = htons(xport);
766 }
767 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to address %s", desc, "ANY");
768 }
769
770 /* Port specified or any */
771 if (xport != 0) {
772 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to port %d", desc, xport);
773 } else {
774 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to port %s", desc, "ANY");
775 }
776
777 /* Set up socket */
778 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: create", desc);
779 if ((xsock = socket(family, SOCK_DGRAM, IPPROTO_UDP)) == -1) {
780 perror("socket");
781 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot create DGRAM socket (%d)", errno);
782
783 exit(EXIT_FAILURE);
784 }
785
786 if (xif != NULL) {
787#ifdef __MACH__
788 unsigned int xif_idx;
789
790 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to interface %s", desc, xif);
791
792 if ((xif_idx = if_nametoindex(xif)) == 0) {
793 perror("if_nametoindex");
794 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot get the interface ID (%d)", errno);
795
796 exit(EXIT_FAILURE);
797 }
798
799 {
800 int if_level = (family == AF_INET6) ? IPPROTO_IPV6 : IPPROTO_IP;
801 int if_optname = (family == AF_INET6) ? IPV6_BOUND_IF : IP_BOUND_IF;
802 if (setsockopt(xsock, if_level, if_optname, &xif_idx, sizeof(xif_idx)) == -1) {
803 perror("setsockopt");
804 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot set socket interface (%d)", errno);
805
806 exit(EXIT_FAILURE);
807 }
808 }
809#elif __unix__
810 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to interface %s", desc, xif);
811
812 if (strlen(xif) >= IFNAMSIZ) {
813 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Interface name too long (max %d): %s", IFNAMSIZ - 1, xif);
814 exit(EXIT_FAILURE);
815 }
816
817 if (setsockopt(xsock, SOL_SOCKET, SO_BINDTODEVICE, xif, strlen(xif) + 1) == -1) {
818 perror("setsockopt");
819 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot set socket interface (%d)", errno);
820
821 exit(EXIT_FAILURE);
822 }
823#endif
824 } else {
825 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind to interface %s", desc, "ANY");
826 }
827
828 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: reuse local address", desc);
829
830 if (setsockopt(xsock, SOL_SOCKET, SO_REUSEADDR, &enable, sizeof(int)) < 0) {
831 perror("setsockopt");
832 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot set socket SO_REUSEADDR (%d)", errno);
833
834 exit(EXIT_FAILURE);
835 }
836
837 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: set nonblocking", desc);
838 if (fcntl(xsock, F_SETFL, O_NONBLOCK) == -1) {
839 perror("fcntl");
840 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot set socket O_NONBLOCK (%d)", errno);
841
842 exit(EXIT_FAILURE);
843 }
844
845 DEBUG(debug_level, DEBUG_LEVEL_INFO, "%s socket: bind", desc);
846 if (bind(xsock, (struct sockaddr *)&addr, addr_len(&addr)) == -1) {
847 perror("bind");
848 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot bind socket (%d)", errno);
849
850 exit(EXIT_FAILURE);
851 }
852
853 socklen_t xsock_name_len = sizeof(*xsock_name);
854 if (getsockname(xsock, (struct sockaddr *)xsock_name, &xsock_name_len) == -1) {
855 perror("getsockname");
856 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Cannot get socket name (%d)", errno);
857
858 exit(EXIT_FAILURE);
859 }
860
861 return xsock;
862}

870static char *resolve_host(int debug_level, const char *host) {
871 struct addrinfo hints;
872 struct addrinfo *res;
873 char buf[INET6_ADDRSTRLEN];
874 char *retval;
875 int rc;
876
877 memset(&hints, 0, sizeof(hints));
878 hints.ai_family = AF_UNSPEC; /* accept IPv4 or IPv6 */
879 hints.ai_socktype = SOCK_DGRAM;
880
881 if ((rc = getaddrinfo(host, NULL, &hints, &res)) != 0) {
882 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Could not resolve host %s: %s", host, gai_strerror(rc));
883
884 exit(EXIT_FAILURE);
885 }
886
887 /* Use the first result; inet_ntop converts it back to a string for parse_addr() */
888 {
889 const void *addr_ptr = (res->ai_family == AF_INET6)
890 ? (const void *)&((struct sockaddr_in6 *)res->ai_addr)->sin6_addr
891 : (const void *)&((struct sockaddr_in *)res->ai_addr)->sin_addr;
892
893 if (inet_ntop(res->ai_family, addr_ptr, buf, sizeof(buf)) == NULL) {
894 perror("inet_ntop");
895 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Could not format resolved address (%d)", errno);
896 freeaddrinfo(res);
897
898 exit(EXIT_FAILURE);
899 }
900 }
901
902 freeaddrinfo(res);
903
904 if ((retval = strdup(buf)) == NULL) {
905 perror("strdup");
906 DEBUG(debug_level, DEBUG_LEVEL_ERROR, "Could not duplicate resolved address string (%d)", errno);
907
908 exit(EXIT_FAILURE);
909 }
910
911 DEBUG(debug_level, DEBUG_LEVEL_DEBUG, "Resolved %s to %s", host, retval);
912
913 return retval;
914}
